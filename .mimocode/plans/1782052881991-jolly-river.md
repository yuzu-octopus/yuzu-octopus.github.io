- [ ] **Step 1: Create ThemeContext.tsx**

```tsx
import { createContext, useContext, useState, useEffect, ReactNode } from 'react';

type Theme = 'dark' | 'light';

interface ThemeContextType {
  theme: Theme;
  toggleTheme: () => void;
}

const ThemeContext = createContext<ThemeContextType>({ theme: 'dark', toggleTheme: () => {} });

export function ThemeProvider({ children }: { children: ReactNode }) {
  const [theme, setTheme] = useState<Theme>(() => {
    return (localStorage.getItem('theme') as Theme) || 'dark';
  });

  useEffect(() => {
    document.documentElement.setAttribute('data-theme', theme);
    localStorage.setItem('theme', theme);
    // Update theme-color meta
    const meta = document.querySelector('meta[name="theme-color"]');
    if (meta) {
      meta.setAttribute('content', theme === 'dark' ? '#282a36' : '#FFFBEB');
    }
  }, [theme]);

  const toggleTheme = () => setTheme(prev => prev === 'dark' ? 'light' : 'dark');

  return (
    <ThemeContext.Provider value={{ theme, toggleTheme }}>
      {children}
    </ThemeContext.Provider>
  );
}

export const useTheme = () => useContext(ThemeContext);
```

## Task 4: Scroll-Reveal Animations

**Covers:** IntersectionObserver-based section animations

**Files:**
- Create: `src/hooks/useScrollReveal.ts`
- Modify: `src/styles/global.css`
- Modify: `src/components/Hero.tsx`, `About.tsx`, `Projects.tsx`, `ConfigsGallery.tsx`, `Workspace.tsx`

**Interfaces:**
- Consumes: `useEffect`, `useRef` from React
- Produces: `useScrollReveal(ref)` hook, `.reveal` / `.visible` CSS classes

- [ ] **Step 1: Create useScrollReveal.ts hook**

```typescript
import { useEffect, useRef } from 'react';

export function useScrollReveal<T extends HTMLElement>() {
  const ref = useRef<T>(null);

  useEffect(() => {
    const el = ref.current;
    if (!el) return;

    const observer = new IntersectionObserver(
      (entries) => {
        entries.forEach((entry) => {
          if (entry.isIntersecting) {
            el.classList.add('visible');
            observer.unobserve(el);
          }
        });
      },
      { threshold: 0.1 }
    );

    observer.observe(el);
    return () => observer.disconnect();
  }, []);

  return ref;
}
```

## Task 8: Sidebar Scroll-Spy

**Covers:** Highlight active nav item based on scroll position

**Files:**
- Create: `src/hooks/useScrollSpy.ts`
- Modify: `src/components/Sidebar.tsx`

**Interfaces:**
- Consumes: IntersectionObserver API
- Produces: `useScrollSpy(sectionIds)` hook returning active section

- [ ] **Step 1: Create useScrollSpy.ts hook**

```typescript
import { useState, useEffect } from 'react';

export function useScrollSpy(ids: string[], offset = 120) {
  const [active, setActive] = useState(ids[0]);

  useEffect(() => {
    const observer = new IntersectionObserver(
      (entries) => {
        entries.forEach((entry) => {
          if (entry.isIntersecting) {
            setActive(entry.target.id);
          }
        });
      },
      { rootMargin: `-${offset}px 0px -40% 0px` }
    );

    ids.forEach((id) => {
      const el = document.getElementById(id);
      if (el) observer.observe(el);
    });

    return () => observer.disconnect();
  }, [ids, offset]);

  return active;
}
```

- [ ] **Step 2: Apply scroll-spy to Sidebar.tsx**
